# 滚动引用诊断

```dataviewjs
// ===== 诊断模式 =====
this.container.innerHTML = '';

// Step 1: 检查数据库页面是否找到
const pages = dv.pages('"09_知识库管理/引用数据库"');
dv.paragraph(`📊 Step1 - 找到页面数: ${pages.length}`);

if (pages.length === 0) {
  dv.paragraph(`❌ 没找到引用数据库！请检查路径 "09_知识库管理/引用数据库"`);
} else {
  const p = pages[0];

  // Step 2: 检查 lists 是否存在
  const lists = p.file.lists;
  dv.paragraph(`📊 Step2 - lists 条数: ${lists ? lists.length : 'null/undefined'}`);

  if (lists && lists.length > 0) {
    // Step 3: 打印前 3 条的结构
    dv.paragraph(`📊 Step3 - 前 3 条 list 结构：`);
    for (let i = 0; i < Math.min(3, lists.length); i++) {
      const item = lists[i];
      dv.paragraph(
        `  [${i}] text="${item.text?.substring(0,50)}" | 类别="${item.类别}" | 语录="${item.语录}" | 作者="${item.作者}" | keys=${JSON.stringify(Object.keys(item))}`
      );
    }

    // Step 4: 尝试构建 q 数组
    const q = pages.flatMap(p => {
      const r = [];
      for (const i of (p.file.lists || [])) {
        if (i.类别 && i.语录) {
          r.push({c: i.类别, t: i.语录, a: (i.作者 && i.作者.trim()) ? i.作者 : null});
        }
      }
      return r;
    });

    dv.paragraph(`📊 Step4 - q 数组长度: ${q.length}`);

    if (q.length > 0) {
      // Step 5: 检查分类
      const categories = [...new Set(q.map(x => x.c))];
      dv.paragraph(`📊 Step5 - 分类: ${categories.join(', ')}`);

      // Step 6: 检查标签匹配
      const t = dv.current().tags || [];
      dv.paragraph(`📊 Step6 - 当前笔记 tags: ${JSON.stringify(t)}`);

      const m = {"交易日志":"交易日志","交易":"交易日志"};
      let cat = null;
      for (const i of t) {
        if (m[i]) { cat = m[i]; break; }
      }
      dv.paragraph(`📊 Step6 - 匹配到的类别: ${cat || 'null'}`);

      const mq = cat ? q.filter(x => x.c === cat) : [];
      dv.paragraph(`📊 Step6 - 匹配到的语录数: ${mq.length}`);

      // Step 7: 显示随机语录
      if (mq.length > 0) {
        const x = mq[Math.floor((Date.now() * 9301 + 49297) % 233280 / 233280 * mq.length)];
        dv.el('blockquote', `${x.t}\n\n— ${x.a || '佚名'}`);
      }
    }
  }
}
```
